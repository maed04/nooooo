Insertion sort

          def insertionSort(A):

          for step in range(1, len(A)):
            key = A[step]
            j = step - 1
            while j >= 0 and key < A[j]:
              A[j + 1] = A[j]
              j = j - 1
              A[j + 1] = key

          data = [9, 5, 1, 4, 3, 999, 45, 456, 123, 0, -12, -1, -50, 16, 55]
          insertionSort(data)
          print(data)
          print("Analisis de algoritmos")
